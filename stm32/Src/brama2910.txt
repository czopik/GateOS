// -------------------- FUNKCJE DLA CZUJNIKA HALLA --------------------
void IRAM_ATTR hallSensorInterrupt() {
  // Odczytaj aktualny kierunek
  // 'direction' jest volatile, więc można go bezpiecznie odczytać w ISR
  if (direction == 1) {
    hallCounter++;
  } else {
    hallCounter--;
    // Zapobieganie ujemnym wartościom hallCounter
    if (hallCounter < 0) {
      hallCounter = 0;
    }
  }
}

// -------------------- FUNKCJE DLA STEROWANIA BRAMĄ --------------------

// Aktualizacja pozycji bramy na podstawie liczby impulsów Halla
void updateGatePosition() {
  noInterrupts();
  long currentHallCounter = hallCounter;
  interrupts();

  static long lastHallCounter = 0;
  long pulses = currentHallCounter - lastHallCounter;

  if (pulses != 0) {
    position += (pulses * gateConfig.wheelCircumference) / gateConfig.pulsesPerRevolution;

    // Zapewnij, że pozycja pozostaje w dozwolonym zakresie
    position = constrain(position, 0.0f, gateConfig.totalDistance);

    lastHallCounter = currentHallCounter;
    debugPrint("Aktualna pozycja: " + String(position) + " m, HallCounter: " + String(currentHallCounter));
  }
}

void startGate(bool openDirection) {
  if (openDirection && digitalRead(LIMIT_SWITCH_OPEN_PIN) == LOW) {  // NC Logic
    debugPrint("Brama jest już otwarta. Nie można otworzyć.");
    return;
  }
  if (!openDirection && digitalRead(LIMIT_SWITCH_CLOSE_PIN) == LOW) { // NC Logic
    debugPrint("Brama jest już zamknięta. Nie można zamknąć.");
    return;
  }

  direction = openDirection ? 1 : -1;
  gateState = openDirection ? GATE_OPENING : GATE_CLOSING;
  isMoving = true;
  gateStartTime = millis();
  currentPWM = gateConfig.pwmMin;

  digitalWrite(MOTOR_IN1_PIN, (direction == 1) ? HIGH : LOW);  // Ustaw kierunek tylko raz na początku
  analogWrite(MOTOR_PWM_PIN, currentPWM);

  debugPrint(openDirection ? "Otwieranie bramy" : "Zamykanie bramy");
}

void stopGate() {
  analogWrite(MOTOR_PWM_PIN, 0);
  isMoving = false;
  gateState = GATE_STOPPED;
  debugPrint("Zatrzymanie bramy");
}

void openGate() {
  if (!isMoving && gateState != GATE_OPEN && gateState != GATE_HOMING) {
    startGate(true);
  }
}

void closeGate() {
  if (!isMoving && gateState != GATE_CLOSED && gateState != GATE_HOMING) {
    startGate(false);
  }
}

void toggleGate() {
  if (gateState == GATE_HOMING) {
    debugPrint("Brama jest w rutynie Home. Nie można wywołać toggle.");
    return;
  }

  if (isMoving) {
    stopGate();
  } else if (gateState == GATE_OPEN || gateState == GATE_OPENING) {
    closeGate();
  } else {
    openGate();
  }
}

// Funkcja centralna do obsługi komend
void processCommand(String command) {
  command.toUpperCase(); // Upewnij się, że komenda jest w wielkich literach
  if (command == "OPEN") {
    openGate();
  } else if (command == "CLOSE") {
    closeGate();
  } else if (command == "TOGGLE") {
    toggleGate();
  } else if (command == "STOP") {
    stopGate();
  } else if (command == "HOME") {
    startHoming();
  } else {
    debugPrint("Nieznana komenda: " + command);
  }
}

void startHoming() {
  // Ustawienie stanu homing
  gateState = GATE_HOMING;
  isMoving = true;
  direction = -1; // Zakładam, że homing polega na zamykaniu
  currentPWM = gateConfig.slowSpeed; // Niska prędkość dla homing

  analogWrite(MOTOR_PWM_PIN, currentPWM);
  digitalWrite(MOTOR_IN1_PIN, (direction == 1) ? HIGH : LOW);
  gateStartTime = millis();
  debugPrint("Rozpoczynanie rutyny Home (kalibracja) - Zamykanie bramy powoli.");
}

void handleGate() {
  unsigned long currentTime = millis();

  if (isMoving) {
    unsigned long elapsedTime = currentTime - gateStartTime;
    float distanceToEnd = (direction == 1) ? (gateConfig.totalDistance - position) : position;

    // Sprawdzenie timeout
    if (elapsedTime > gateConfig.movementTimeout) {
      stopGate();
      gateState = GATE_ERROR;
      debugPrint("Timeout: Brama nie osiągnęła krańcówki w oczekiwanym czasie.");
      // Można dodać powiadomienie przez MQTT lub WebSocket
      return;
    }

    if (gateState == GATE_HOMING) {
      // Homing: Poruszanie się powoli w kierunku zamknięcia
      currentPWM = gateConfig.slowSpeed; // Możesz ustawić jeszcze niższą wartość PWM, jeśli potrzebujesz

      analogWrite(MOTOR_PWM_PIN, currentPWM);
      digitalWrite(MOTOR_IN1_PIN, (direction == 1) ? HIGH : LOW);

      // Sprawdź krańcówkę zamknięcia
      bool limitSwitchCloseActive = digitalRead(LIMIT_SWITCH_CLOSE_PIN) == LOW; // NC logic

      if (limitSwitchCloseActive) {
        // Zatrzymaj bramę
        stopGate();
        gateState = GATE_CLOSED;
        position = 0;
        hallCounter = 0;
        debugPrint("Rutyna Home zakończona. Brama jest zamknięta.");
      }
      return; // Nie wykonuj dalszej logiki dla homing
    }

    // Soft Start
    if (gateFlags.softStartEnabled && elapsedTime < gateConfig.softStartTime) {
      float progress = static_cast<float>(elapsedTime) / gateConfig.softStartTime;
      currentPWM = gateConfig.pwmMin + (gateConfig.pwmMax - gateConfig.pwmMin) * progress;
    } else {
      currentPWM = gateConfig.pwmMax;
    }

    // Soft Stop
    if (gateFlags.softStopEnabled && distanceToEnd <= gateConfig.slowDownDistance) {
      // Zmniejsz prędkość proporcjonalnie do odległości do końca
      float progress = distanceToEnd / gateConfig.slowDownDistance;
      currentPWM = gateConfig.pwmMin + (gateConfig.pwmMax - gateConfig.pwmMin) * progress;
      currentPWM = constrain(currentPWM, gateConfig.pwmMin, gateConfig.pwmMax);
    }

    // Ustaw PWM i kierunek silnika
    analogWrite(MOTOR_PWM_PIN, currentPWM);
    digitalWrite(MOTOR_IN1_PIN, (direction == 1) ? HIGH : LOW);

    // Sprawdź krańcówki
    bool limitSwitchOpenActive = digitalRead(LIMIT_SWITCH_OPEN_PIN) == LOW;  // NC logic
    bool limitSwitchCloseActive = digitalRead(LIMIT_SWITCH_CLOSE_PIN) == LOW; // NC logic

    // Warunek zatrzymania bramy
    if ((direction == 1 && (limitSwitchOpenActive || position >= gateConfig.totalDistance)) ||
        (direction == -1 && (limitSwitchCloseActive || position <= 0))) {

      // Zatrzymaj silnik
      stopGate();

      // Aktualizacja stanu bramy
      if (direction == 1) {
        gateState = GATE_OPEN;
        position = gateConfig.totalDistance;
        debugPrint("Bramę otwarto. Pozycja: " + String(position));
      } else {
        gateState = GATE_CLOSED;
        position = 0;
        debugPrint("Bramę zamknięto. Pozycja: " + String(position));
      }

      debugPrint("Bramę zatrzymano.");
    }
  } else {
    // Gdy brama nie się porusza, upewnij się, że silnik jest wyłączony
    analogWrite(MOTOR_PWM_PIN, 0);
  }
}

// -------------------- FUNKCJE DLA PILOTA HCS301 --------------------
void IRAM_ATTR HCS_interrupt() {
  if (!HCS_Listening) {
    return;
  }

  uint32_t cur_timestamp = micros();
  uint8_t cur_status = digitalRead(HCS_RECEIVER_PIN);
  uint32_t pulse_duration = cur_timestamp - HCS_last_change;
  HCS_last_change = cur_timestamp;

  if (bPreamble) {
    if (pulse_duration > 3000 && pulse_duration < 6000) {
      bHeader = true;
      HCS_bit_counter = 0;
      bPreamble = false;
      bInv = cur_status == LOW;
    } else if (pulse_duration < 300 || pulse_duration > 600) {
      bPreamble = false;
      digitalWrite(LED_INDICATOR_PIN, HIGH);
    }
  } else if (bHeader) {
    if ((pulse_duration > 300) && (pulse_duration < 1100)) {
      if (cur_status == (bInv ? HIGH : LOW)) {
        if (HCS_bit_counter < 66) {
          HCS_bit_array[65 - HCS_bit_counter] = (pulse_duration > HCS_Te2_3) ? 0 : 1;
          HCS_bit_counter++;
          if (HCS_bit_counter == 66) {
            bHeader = false;
            digitalWrite(LED_INDICATOR_PIN, HIGH);
            HCS_Listening = false;
            HCS_preamble_count = 0;

            // Skopiuj dane do globalnej struktury hcs301
            hcs301.Repeat = HCS_bit_array[0];
            hcs301.BattaryLow = HCS_bit_array[1];
            hcs301.BtnToggle = HCS_bit_array[3]; // Czerwony przycisk
            hcs301.BtnGreen = HCS_bit_array[4];  // Zielony przycisk

            hcs301.SerialNum = 0;
            for (int i = 5; i < 33; i++) {
              hcs301.SerialNum = (hcs301.SerialNum << 1) + HCS_bit_array[i];
            }

            uint32_t Encript = 0;
            for (int i = 33; i < 66; i++) {
              Encript = (Encript << 1) + HCS_bit_array[i];
            }
            hcs301.Encript = Encript;

            debugPrint("Odebrano pilota: SerialNum=" + String(hcs301.SerialNum) + ", Encript=" + String(hcs301.Encript));
          }
        }
      }
    } else {
      bHeader = false;
    }
  } else {
    if (pulse_duration > 300 && pulse_duration < 600) {
      if (++HCS_preamble_count > 10) {
        bPreamble = true;
        HCS_preamble_count = 0;
      }
    } else {
      HCS_preamble_count = 0;
    }
  }
}

bool isAuthorizedRemote(unsigned long serialNum) {
  for (int i = 0; i < numAllowedRemotes; i++) {
    if (allowedRemotes[i] == serialNum) {
      return true;
    }
  }
  return false;
}

void handleRemoteSignal() {
  if (!HCS_Listening) {
    // Przetwarzaj odebrany sygnał z pilota
    noInterrupts();
    HCS301 msg = hcs301;
    HCS_Listening = true;
    interrupts();

    msg.print();

    // Publikowanie danych do MQTT
    char payload[256];
    snprintf(payload, sizeof(payload), "{\"SerialNum\":%lu,\"Buttons\":\"%s\",\"BatteryLow\":%d,\"Repeat\":%d}",
             msg.SerialNum,
             msg.BtnGreen ? "Green " : "",
             msg.BattaryLow,
             msg.Repeat);
    mqttClient.publish("home/HCS301", payload);

    digitalWrite(LED_INDICATOR_PIN, HIGH);

    if (isAuthorizedRemote(msg.SerialNum)) {
      debugPrint("Pilot autoryzowany: " + String(msg.SerialNum));

      int remoteIndex = -1;
      for (int i = 0; i < numAllowedRemotes; i++) {
        if (allowedRemotes[i] == msg.SerialNum) {
          remoteIndex = i;
          break;
        }
      }

      if (remoteIndex != -1) {
        if (lastEncript[remoteIndex] != msg.Encript && msg.BtnGreen) {
          processCommand("TOGGLE"); // Użycie centralnej funkcji
          debugPrint("toggleGate() wywołane przez pilota RF " + String(msg.SerialNum));
          lastEncript[remoteIndex] = msg.Encript;
        }
      }
    } else {
      lastReceivedSerialNum = msg.SerialNum;
      lastReceivedButtons = "";
      if (msg.BtnToggle) lastReceivedButtons += "Toggle ";
      if (msg.BtnGreen) lastReceivedButtons += "Green ";
      analogWrite(MOTOR_PWM_PIN, 0);
    }
  }
}

// -------------------- FUNKCJE DLA MQTT --------------------
void mqttCallback(char* topic, byte* payload, unsigned int length) {
  String incomingTopic = String(topic);
  String message;

  for (unsigned int i = 0; i < length; i++) {
    message += (char)payload[i];
  }

  Serial.print("Odebrano wiadomość na temacie: ");
  debugPrint(incomingTopic);
  Serial.print("Wiadomość: ");
  debugPrint(message);

  if (incomingTopic == mqtt_topic_gate_command) {
    processCommand(message); // Użycie centralnej funkcji
  }
}

void reconnectMQTT() {
  while (!mqttClient.connected()) {
    Serial.print("Łączenie z MQTT...");
    String clientId = "ESP8266Client-" + String(ESP.getChipId(), HEX);
    if (mqttClient.connect(clientId.c_str(), mqtt_user, mqtt_password)) {
      debugPrint("Połączono z MQTT");
      // Subskrybuj temat sterowania
      mqttClient.subscribe(mqtt_topic_gate_command);
    } else {
      Serial.print("Nieudane, rc=");
      Serial.print(mqttClient.state());
      debugPrint(" próbuję ponownie za 5 sekund");
      delay(5000);
    }
  }
}

// Wysyłanie aktualizacji statusu przez WebSocket
void sendStatusUpdate() {
  StaticJsonDocument<256> jsonDoc;
  jsonDoc["type"] = "status";
  jsonDoc["position"] = position;
  jsonDoc["limitSwitchOpen"] = (digitalRead(LIMIT_SWITCH_OPEN_PIN) == LOW);
  jsonDoc["limitSwitchClose"] = (digitalRead(LIMIT_SWITCH_CLOSE_PIN) == LOW);
  jsonDoc["isMoving"] = isMoving;
  jsonDoc["gateStatus"] = gateStatus;
  jsonDoc["hallCounter"] = hallCounter; // Liczba impulsów Halla

  String jsonString;
  serializeJson(jsonDoc, jsonString);

  // Publikowanie statusu bramy do MQTT
  mqttClient.publish(mqtt_topic_gate_state, jsonString.c_str());

  // Wysyłanie statusu do wszystkich klientów WebSocket
  ws.textAll(jsonString);
}

// Handler zdarzeń WebSocket
void onWebSocketEvent(AsyncWebSocket * server, AsyncWebSocketClient * client, AwsEventType type, void * arg, uint8_t *data, size_t len) {
  if (type == WS_EVT_CONNECT) {
    // Wyślij początkowy status
    sendStatusUpdate();
  } else if (type == WS_EVT_DISCONNECT) {
    // Możesz dodać logikę przy rozłączeniu klienta, jeśli potrzebujesz
  } else if (type == WS_EVT_DATA) {
    // Obsługa danych odebranych od klienta
    AwsFrameInfo *info = (AwsFrameInfo*)arg;
    if (info->final && info->index == 0 && info->len == len) {
      // Cała wiadomość jest w jednym ramce
      String msg = "";
      for (size_t i = 0; i < len; i++) {
        msg += (char)data[i];
      }
      // Przetwórz wiadomość, jeśli to konieczne
      debugPrint("Odebrano wiadomość przez WebSocket: " + msg);
      processCommand(msg); // Opcjonalnie, jeśli chcesz umożliwić sterowanie przez WebSocket
    }
  }
}

void handleSetConfig(AsyncWebServerRequest *request, uint8_t *data, size_t len, size_t index, size_t total) {
  static String configBody = "";

  // Jeśli zaczynamy nowe żądanie, resetujemy bufor
  if (index == 0) {
    configBody = "";
  }

  // Dodaj aktualny fragment danych do bufora
  // Używamy tymczasowego bufora, aby uniknąć problemów z konstrukcją String
  char temp[len + 1];
  memcpy(temp, data, len);
  temp[len] = '\0';
  configBody += String(temp);

  // Jeśli odebraliśmy wszystkie dane, przetwarzamy je
  if (index + len >= total) {
    debugPrint("Odebrano pełny body: " + configBody);

    DynamicJsonDocument doc(1024);
    DeserializationError error = deserializeJson(doc, configBody);
    if (error) {
      debugPrint("Błąd deserializacji JSON: " + String(error.c_str()));
      request->send(400, "application/json", "{\"status\":\"error\",\"message\":\"Invalid JSON\"}");
      return;
    }

    // Aktualizacja zmiennych globalnych z GateConfig
    gateConfig.totalDistance = doc["totalDistance"] | gateConfig.totalDistance;
    gateConfig.wheelCircumference = doc["wheelCircumference"] | gateConfig.wheelCircumference;
    gateConfig.pulsesPerRevolution = doc["pulsesPerRevolution"] | gateConfig.pulsesPerRevolution;
    gateConfig.gateSpeed = doc["gateSpeed"] | gateConfig.gateSpeed;
    gateConfig.slowSpeed = doc["slowSpeed"] | gateConfig.slowSpeed;
    gateConfig.slowDownDistance = doc["slowDownDistance"] | gateConfig.slowDownDistance;
    gateConfig.softStartTime = doc["softStartTime"] | gateConfig.softStartTime;
    gateConfig.pwmMin = doc["pwmMin"] | gateConfig.pwmMin;
    gateConfig.pwmMax = doc["pwmMax"] | gateConfig.pwmMax;
    gateConfig.movementTimeout = doc["movementTimeout"] | gateConfig.movementTimeout;

    // Aktualizacja flag
    gateFlags.softStartEnabled = doc["softStartEnabled"] | gateFlags.softStartEnabled;
    gateFlags.softStopEnabled = doc["softStopEnabled"] | gateFlags.softStopEnabled;
    gateFlags.debugMode = doc["debugMode"] | gateFlags.debugMode;
    gateFlags.newRemoteLearnMode = doc["newRemoteLearnMode"] | gateFlags.newRemoteLearnMode;

    // Zapisz ustawienia do LittleFS
    saveSettings();

    // Potwierdzenie sukcesu
    request->send(200, "application/json", "{\"status\":\"success\"}");
    configBody = ""; // Reset bufora po przetworzeniu
  }
}

void handleControl(AsyncWebServerRequest *request) {
  if (request->hasParam("command")) {
    String command = request->getParam("command")->value();
    processCommand(command); // Użycie centralnej funkcji
  }
  request->send(200, "text/plain", "OK");
}

void handleAddRemote(AsyncWebServerRequest *request) {
  if (request->hasParam("serialNum", true)) {
    // Pobierz wartość parametru 'serialNum' jako String
    String serialStr = request->getParam("serialNum", true)->value();
    debugPrint("Received serialNum: " + serialStr);

    // Konwertuj String na unsigned long za pomocą strtoul()
    unsigned long serialNum = strtoul(serialStr.c_str(), NULL, 10);
    debugPrint("Converted serialNum: " + String(serialNum));

    // Sprawdź, czy konwersja się powiodła
    if (serialNum == 0 && serialStr != "0") {
      debugPrint("Invalid Serial Number");
      request->send(400, "application/json", "{\"status\":\"error\",\"message\":\"Invalid Serial Number\"}");
      return;
    }

    // Dodaj pilota do listy, jeśli nie przekroczono limitu
    if (numAllowedRemotes < MAX_REMOTES) {
      allowedRemotes[numAllowedRemotes] = serialNum;
      lastEncript[numAllowedRemotes] = 0; // Inicjalizacja
      numAllowedRemotes++;
      saveRemotes();
      lastReceivedSerialNum = 0; // Reset po dodaniu
      debugPrint("Added new remote: " + String(serialNum));
      request->redirect("/remotes.html");
    } else {
      debugPrint("Maximum number of remotes reached");
      request->send(500, "text/plain", "Osiągnięto maksymalną liczbę pilotów.");
    }
  } else {
    debugPrint("No serialNum parameter found");
    request->send(400, "text/plain", "Brak numeru seryjnego pilota.");
  }
}

void handleGetRemotes(AsyncWebServerRequest *request) {
  DynamicJsonDocument doc(1024);
  JsonArray remotes = doc.createNestedArray("remotes");
  for (int i = 0; i < numAllowedRemotes; i++) {
    remotes.add(allowedRemotes[i]);
  }
  String json;
  serializeJson(doc, json);
  request->send(200, "application/json", json);
}

// Ustawienie serwera webowego
void setupWebServer() {
  // Serwowanie plików statycznych z LittleFS
  server.serveStatic("/", LittleFS, "/").setDefaultFile("index.html");

  // Trasy API
  server.on("/config", HTTP_GET, handleGetConfig);

  // Poprawiona obsługa POST z body
  server.on("/config", HTTP_POST, [](AsyncWebServerRequest *request){
    // Do nothing tutaj
  }, NULL, handleSetConfig);

  server.on("/control", HTTP_GET, handleControl);
  server.on("/add_remote", HTTP_POST, handleAddRemote);
  server.on("/get_config_file", HTTP_GET, handleGetConfigFile);
  server.on("/remotes", HTTP_GET, handleGetRemotes);

  // Obsługa nieznalezionych stron
  server.onNotFound(handleNotFound);

  // Dodaj WebSocket
  ws.onEvent(onWebSocketEvent);
  server.addHandler(&ws);

  server.begin();
  debugPrint("Serwer webowy uruchomiony.");
}

void loadSettings() {
  if (!LittleFS.exists(CONFIG_PATH)) {
    Serial.println("Brak pliku z ustawieniami, używam domyślnych");
    return;
  }
  File configFile = LittleFS.open(CONFIG_PATH, "r");
  if (!configFile) {
    Serial.println("Nie można otworzyć pliku z ustawieniami");
    return;
  }
  size_t size = configFile.size();
  if (size > 1024) { // Zwiększony rozmiar bufora
    Serial.println("Plik z ustawieniami jest za duży");
    configFile.close();
    return;
  }
  DynamicJsonDocument doc(1024);
  DeserializationError error = deserializeJson(doc, configFile);
  configFile.close();
  if (error) {
    Serial.println("Błąd podczas odczytu pliku z ustawieniami");
    return;
  }
  gateConfig.totalDistance = doc["totalDistance"] | gateConfig.totalDistance;
  gateConfig.wheelCircumference = doc["wheelCircumference"] | gateConfig.wheelCircumference;
  gateConfig.pulsesPerRevolution = doc["pulsesPerRevolution"] | gateConfig.pulsesPerRevolution;
  gateConfig.gateSpeed = doc["gateSpeed"] | gateConfig.gateSpeed;
  gateConfig.slowSpeed = doc["slowSpeed"] | gateConfig.slowSpeed;
  gateConfig.slowDownDistance = doc["slowDownDistance"] | gateConfig.slowDownDistance;
  gateConfig.softStartTime = doc["softStartTime"] | gateConfig.softStartTime;
  gateConfig.pwmMin = doc["pwmMin"] | gateConfig.pwmMin;
  gateConfig.pwmMax = doc["pwmMax"] | gateConfig.pwmMax;
  gateConfig.movementTimeout = doc["movementTimeout"] | gateConfig.movementTimeout;

  // Flagi
  gateFlags.softStartEnabled = doc["softStartEnabled"] | gateFlags.softStartEnabled;
  gateFlags.softStopEnabled = doc["softStopEnabled"] | gateFlags.softStopEnabled;
  gateFlags.debugMode = doc["debugMode"] | gateFlags.debugMode;
  gateFlags.newRemoteLearnMode = doc["newRemoteLearnMode"] | gateFlags.newRemoteLearnMode;

  debugPrint("Ustawienia zostały załadowane.");
}

// Funkcje do ładowania i zapisywania listy pilotów
void loadRemotes() {
  if (!LittleFS.exists("/remotes.json")) {
    debugPrint("Plik remotes.json nie istnieje, lista będzie pusta.");
    return;
  }

  File file = LittleFS.open("/remotes.json", "r");
  if (!file) {
    debugPrint("Nie można otworzyć pliku remotes.json, lista będzie pusta.");
    return;
  }

  size_t size = file.size();
  if (size == 0) {
    debugPrint("Plik remotes.json jest pusty.");
    file.close();
    return;
  }

  std::unique_ptr<char[]> buf(new char[size]);

  file.readBytes(buf.get(), size);
  file.close();

  StaticJsonDocument<1024> doc;
  DeserializationError error = deserializeJson(doc, buf.get());
  if (error) {
    debugPrint("Błąd podczas parsowania pliku remotes.json.");
    return;
  }

  numAllowedRemotes = 0;
  JsonArray array = doc["remotes"];
  if (array.isNull()) {
    debugPrint("Brak pilotów w pliku remotes.json.");
    return;
  }

  for (JsonVariant v : array) {
    if (numAllowedRemotes < MAX_REMOTES) {
      allowedRemotes[numAllowedRemotes] = v.as<unsigned long>();
      lastEncript[numAllowedRemotes] = 0;
      debugPrint("Załadowano pilota o numerze seryjnym: " + String(allowedRemotes[numAllowedRemotes]));
      numAllowedRemotes++;
    } else {
      debugPrint("Osiągnięto maksymalną liczbę pilotów.");
      break;
    }
  }

  debugPrint("Lista autoryzowanych pilotów załadowana.");
}

void saveRemotes() {
  StaticJsonDocument<1024> doc;
  JsonArray array = doc.createNestedArray("remotes");
  for (int i = 0; i < numAllowedRemotes; i++) {
    array.add(allowedRemotes[i]);
  }

  File file = LittleFS.open("/remotes.json", "w");
  if (!file) {
    debugPrint("Nie można zapisać pliku remotes.json.");
    return;
  }

  bool success = serializeJson(doc, file);
  file.close();

  if (!success) {
    debugPrint("Błąd serializacji JSON do pliku remotes.json");
  } else {
    debugPrint("Lista autoryzowanych pilotów zapisana do pliku.");
  }
}

// -------------------- INICJALIZACJA --------------------
void setup() {
  Serial.begin(115200);
  delay(10);

  // Inicjalizacja pinów
  pinMode(LIMIT_SWITCH_OPEN_PIN, INPUT_PULLUP);
  pinMode(LIMIT_SWITCH_CLOSE_PIN, INPUT_PULLUP);
  pinMode(MOTOR_IN1_PIN, OUTPUT);
  pinMode(MOTOR_PWM_PIN, OUTPUT);
  pinMode(HALL_SENSOR_PIN, INPUT_PULLUP);
  pinMode(BUTTON_PIN, INPUT_PULLUP);
  pinMode(HCS_RECEIVER_PIN, INPUT);
  pinMode(LED_INDICATOR_PIN, OUTPUT);
  digitalWrite(LED_INDICATOR_PIN, HIGH);

  // Inicjalizacja debouncerów
  limitSwitchOpenDebouncer.attach(LIMIT_SWITCH_OPEN_PIN);
  limitSwitchOpenDebouncer.interval(50);

  limitSwitchCloseDebouncer.attach(LIMIT_SWITCH_CLOSE_PIN);
  limitSwitchCloseDebouncer.interval(50);

  buttonDebouncer.attach(BUTTON_PIN);
  buttonDebouncer.interval(50);

  // Inicjalizacja czujnika Halla
  // Możesz rozważyć użycie trybu CHANGE zamiast FALLING, jeśli czujnik generuje impulsy zarówno na narastające, jak i opadające zbocze
  attachInterrupt(digitalPinToInterrupt(HALL_SENSOR_PIN), hallSensorInterrupt, FALLING);

  // Inicjalizacja LittleFS
  if (!LittleFS.begin()) {
    debugPrint("Nie udało się zainicjalizować LittleFS");
    return;
  }

  // Wczytanie ustawień
  loadSettings();
  loadRemotes();

  // Połączenie z Wi-Fi
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);
  Serial.println("Łączenie z Wi-Fi...");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("");
  Serial.print("Połączono z siecią Wi-Fi: ");
  Serial.println(WiFi.localIP());

  // Inicjalizacja MQTT
  mqttClient.setServer(mqtt_server, mqtt_port);
  mqttClient.setCallback(mqttCallback);

  // Inicjalizacja OTA
  ArduinoOTA.setPassword("chemik1");
  ArduinoOTA.onStart([]() {
    debugPrint("Rozpoczęcie aktualizacji OTA");
  });
  ArduinoOTA.onEnd([]() {
    debugPrint("Aktualizacja OTA zakończona");
  });
  ArduinoOTA.onError([](ota_error_t error) {
    debugPrint("Błąd OTA");
  });
  ArduinoOTA.begin();
  debugPrint("OTA gotowe");

  // Inicjalizacja serwera webowego
  setupWebServer();

  // Inicjalizacja przerwań
  attachInterrupt(digitalPinToInterrupt(HCS_RECEIVER_PIN), HCS_interrupt, CHANGE);

  // Rutyna Home: kalibracja bramy na starcie
  // Sprawdź, czy brama jest w znanym stanie
  bool openActive = (digitalRead(LIMIT_SWITCH_OPEN_PIN) == LOW);
  bool closeActive = (digitalRead(LIMIT_SWITCH_CLOSE_PIN) == LOW);

  if (!openActive && !closeActive) {
    // Brama nie jest w znanym stanie, rozpocznij rutynę Home
    debugPrint("Brama nie jest w znanym stanie. Rozpoczynam rutynę Home.");
    startHoming();
  } else if (closeActive) {
    // Brama jest zamknięta
    gateState = GATE_CLOSED;
    position = 0;
    hallCounter = 0;
    debugPrint("Brama jest zamknięta.");
  } else if (openActive) {
    // Brama jest otwarta
    gateState = GATE_OPEN;
    position = gateConfig.totalDistance;
    hallCounter = static_cast<long>(gateConfig.totalDistance / gateConfig.wheelCircumference * gateConfig.pulsesPerRevolution);
    debugPrint("Brama jest otwarta.");
  } else {
    // Nieznany stan, rozpocznij rutynę Home
    debugPrint("Nieznany stan bramy. Rozpoczynam rutynę Home.");
    startHoming();
  }
}

// -------------------- PĘTLA GŁÓWNA --------------------
void loop() {
  ArduinoOTA.handle();
  if (!mqttClient.connected()) {
    reconnectMQTT();
  }
  mqttClient.loop();

  limitSwitchOpenDebouncer.update();
  limitSwitchCloseDebouncer.update();
  buttonDebouncer.update();

  handleRemoteSignal();
  handleGate();
  updateGatePosition();  // Wywołanie funkcji aktualizującej pozycję

  // Aktualizacja statusu bramy
  String currentGateStatus = gateStatus;
  if (isMoving) {
    gateStatus = "ruch";
  } else if (digitalRead(LIMIT_SWITCH_CLOSE_PIN) == LOW) {
    gateStatus = "zamknięta";
  } else if (digitalRead(LIMIT_SWITCH_OPEN_PIN) == LOW) {
    gateStatus = "otwarta";
  } else {
    gateStatus = "zatrzymana";
  }

  // Ograniczenie częstotliwości wysyłania aktualizacji
  static unsigned long lastStatusUpdateTime = 0;
  const unsigned long STATUS_UPDATE_INTERVAL = 100; // Aktualizacja co 100 ms
  unsigned long currentTime = millis();

  if (currentTime - lastStatusUpdateTime >= STATUS_UPDATE_INTERVAL) {
    lastStatusUpdateTime = currentTime;

    // Aktualizacja statusu bramy do MQTT i WebSocket
    if (gateStatus != previousGateStatus || positionChanged) {
      sendStatusUpdate();
      previousGateStatus = gateStatus;
      positionChanged = false;
    }
  }

  // Obsługa przycisku lokalnego z debouncingiem
  static bool lastButtonState = HIGH;
  bool buttonState = buttonDebouncer.read();
  if (buttonState != lastButtonState) {
    if (buttonState == LOW) {
      processCommand("TOGGLE"); // Użycie centralnej funkcji
    }
    lastButtonState = buttonState;
  }
}
